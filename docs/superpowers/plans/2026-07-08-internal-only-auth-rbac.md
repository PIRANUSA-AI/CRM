**Goal:** Turn off public self-signup, bootstrap one CEO account via a seed script, and let CEO/Leader create and manage `ceo`/`leader`/`sales` team accounts through a guarded endpoint and a small dedicated admin page.

**Architecture:** Reuse `AgentService.createAgent`/`updateAgent` (already handles password hashing, the Better Auth credential link, and organization membership) behind a new `requireRole()` guard. Block the public Better Auth signup HTTP path at the existing `betterAuthView` wrapper. Add a small new frontend page consuming the existing `agentsManagement` API client, independent of `team.tsx`.

**Tech Stack:** Existing stack, unchanged — Bun, Elysia, Prisma/PostgreSQL, Better Auth, TanStack Start/React frontend.

## Global Constraints

- Do not modify `apps/frontend/src/routes/_app/team.tsx` or `apps/backend/src/modules/handover/service.ts` — their `agent`/`supervisor`/`admin` role values are a separate, unrelated concept and are out of scope.
- `role` stays a plain `String` column (no new Prisma enum/migration) — canonical values `ceo` / `leader` / `sales` are enforced at the API/UI layer only, via one shared constant.
- Every new/modified endpoint in this plan must go through `requireRole()` — no ad-hoc role string comparisons.
- Passwords generated by the system are returned in the HTTP response exactly once and never logged (no `console.log` of request bodies containing a password on this path).
- Login (`/auth/sign-in/email`) must keep working unmodified — only `/auth/sign-up/email` is blocked.

```typescript
// apps/backend/src/lib/require-role.ts
import prisma from './prisma'

export const CANONICAL_ROLES = ['ceo', 'leader', 'sales'] as const
export type CanonicalRole = (typeof CANONICAL_ROLES)[number]

type RequireRoleResult =
	| { ok: true; role: string }
	| { ok: false; status: number; error: string }

export async function requireRole(
	userId: string | null,
	allowedRoles: CanonicalRole[],
): Promise<RequireRoleResult> {
	if (!userId) {
		return { ok: false, status: 403, error: 'Not authenticated' }
	}

	const user = await prisma.users.findUnique({
		where: { id: userId },
		select: { role: true },
	})

	if (!user || !allowedRoles.includes(user.role as CanonicalRole)) {
		return { ok: false, status: 403, error: 'Not authorized' }
	}

	return { ok: true, role: user.role as string }
}
```

```typescript
// apps/backend/src/lib/generate-password.ts
import { randomInt } from 'node:crypto'

const LOWER = 'abcdefghijkmnpqrstuvwxyz'
const UPPER = 'ABCDEFGHJKLMNPQRSTUVWXYZ'
const DIGITS = '23456789'
const SYMBOLS = '!@#$%^&*'
const ALL = LOWER + UPPER + DIGITS + SYMBOLS

function pick(chars: string): string {
	return chars[randomInt(chars.length)]
}

function shuffle<T>(items: T[]): T[] {
	for (let i = items.length - 1; i > 0; i--) {
		const j = randomInt(i + 1)
		;[items[i], items[j]] = [items[j], items[i]]
	}
	return items
}

export function generateStrongPassword(): string {
	const required = [pick(LOWER), pick(UPPER), pick(DIGITS), pick(SYMBOLS)]
	const rest = Array.from({ length: 12 }, () => pick(ALL))

	return shuffle([...required, ...rest]).join('')
}
```

> Corrected during execution: the first version of this task used
> `Math.random()`, which is not cryptographically secure and unsuitable for
> generating login credentials. An automated security review caught it
> post-commit; fixed with `node:crypto`'s `randomInt` in commit `77a5dbe`
> and updated here so a future re-dispatch doesn't reintroduce it.

```typescript
const betterAuthView = async (request: Request) => {
	const BETTER_AUTH_ACCEPT_METHODS = ['POST', 'GET']
	if (BETTER_AUTH_ACCEPT_METHODS.includes(request.method)) {
		const url = new URL(request.url)
		if (request.method === 'POST' && url.pathname.endsWith('/sign-up/email')) {
			return Response.json(
				{ message: 'Public sign-up is disabled', code: 'SIGNUP_DISABLED' },
				{ status: 403 },
			)
		}
		try {
			return await auth.handler(request)
		} catch (e: any) {
			console.error('[AUTH ERROR]', e?.message)
			console.error('[AUTH ERROR STACK]', e?.stack?.substring(0, 500))
			console.error('[AUTH ERROR META]', JSON.stringify(e?.meta))
			return Response.json(
				{
					message: e?.message || 'Auth error',
					code: 'AUTH_ERROR',
					meta: e?.meta,
				},
				{ status: 500 },
			)
		}
	} else {
		return Response.json({ message: 'Method Not Allowed' }, { status: 405 })
	}
}
```

```typescript
export function getAllowedPrimaryPathsForRole(
	role: string | null | undefined,
): string[] | null {
	const normalizedRole = normalizeAppRole(role)

	if (normalizedRole === 'agent' || normalizedRole === 'sales') {
		return ['/dashboard', '/chat', '/channels/whatsapp']
	}

	if (normalizedRole === 'supervisor' || normalizedRole === 'leader') {
		return [
			'/dashboard',
			'/chat',
			'/team',
			'/kelola-tim',
			'/channels/whatsapp',
			'/orders',
			'/products',
			'/product-stock',

			'/settings',
		]
	}

	return null
}
```

```tsx
import { createFileRoute } from '@tanstack/react-router'
import { useEffect, useState } from 'react'
import { agentsManagement } from '@/lib/agents-api'
import { extractNormalizedRole } from '@/lib/role-access'
import { Button } from '@/components/ui/button'
import { Input } from '@/components/ui/input'
import { NativeSelect, NativeSelectOption } from '@/components/ui/native-select'
import { toast } from 'sonner'

export const Route = createFileRoute('/_app/kelola-tim')({
	component: KelolaTimPage,
})

interface TeamMember {
	id: string
	name: string
	email: string
	role: string
}

function KelolaTimPage() {
	const [members, setMembers] = useState<TeamMember[]>([])
	const [name, setName] = useState('')
	const [email, setEmail] = useState('')
	const [role, setRole] = useState('sales')
	const [generatedPassword, setGeneratedPassword] = useState<string | null>(null)
	const [loading, setLoading] = useState(false)

	async function loadMembers() {
		const response = await agentsManagement.list()
		setMembers(response.data ?? [])
	}

	useEffect(() => {
		loadMembers()
	}, [])

	async function handleCreate(e: React.FormEvent) {
		e.preventDefault()
		setLoading(true)
		try {
			const response = await agentsManagement.create({ name, email, role })
			if (response.data?.generatedPassword) {
				setGeneratedPassword(response.data.generatedPassword)
			}
			setName('')
			setEmail('')
			await loadMembers()
			toast.success('Akun berhasil dibuat')
		} catch (err: any) {
			toast.error(err?.message || 'Gagal membuat akun')
		} finally {
			setLoading(false)
		}
	}

	async function handleRoleChange(memberId: string, newRole: string) {
		await agentsManagement.update(memberId, { role: newRole })
		await loadMembers()
	}

	return (
		<div className="p-6 space-y-6">
			<h1 className="text-2xl font-bold">Kelola Tim</h1>

			{generatedPassword && (
				<div className="rounded-md border border-amber-400 bg-amber-50 p-4">
					<p className="text-sm font-medium">
						Password akun baru (catat sekarang, tidak akan ditampilkan lagi):
					</p>
					<code className="mt-1 block text-lg">{generatedPassword}</code>
					<Button
						variant="ghost"
						size="sm"
						onClick={() => setGeneratedPassword(null)}
					>
						Tutup
					</Button>
				</div>
			)}

			<form onSubmit={handleCreate} className="flex flex-wrap gap-2 items-end">
				<Input
					placeholder="Nama"
					value={name}
					onChange={(e) => setName(e.target.value)}
					required
				/>
				<Input
					type="email"
					placeholder="Email"
					value={email}
					onChange={(e) => setEmail(e.target.value)}
					required
				/>
				<NativeSelect value={role} onChange={(e) => setRole(e.target.value)}>
					<NativeSelectOption value="sales">Sales</NativeSelectOption>
					<NativeSelectOption value="leader">Sales Leader</NativeSelectOption>
					<NativeSelectOption value="ceo">CEO</NativeSelectOption>
				</NativeSelect>
				<Button type="submit" disabled={loading}>
					Tambah Akun
				</Button>
			</form>

			<table className="w-full text-sm">
				<thead>
					<tr className="text-left border-b">
						<th className="py-2">Nama</th>
						<th>Email</th>
						<th>Role</th>
					</tr>
				</thead>
				<tbody>
					{members.map((member) => (
						<tr key={member.id} className="border-b">
							<td className="py-2">{member.name}</td>
							<td>{member.email}</td>
							<td>
								<NativeSelect
									value={extractNormalizedRole(member) || member.role}
									onChange={(e) => handleRoleChange(member.id, e.target.value)}
								>
									<NativeSelectOption value="sales">Sales</NativeSelectOption>
									<NativeSelectOption value="leader">Sales Leader</NativeSelectOption>
									<NativeSelectOption value="ceo">CEO</NativeSelectOption>
								</NativeSelect>
							</td>
						</tr>
					))}
				</tbody>
			</table>
		</div>
	)
}
```
